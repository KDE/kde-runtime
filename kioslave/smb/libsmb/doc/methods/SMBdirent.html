class SMBdirent

Class Description

class SMBdirent
{
public:
/* protection */
    uint32 st_mode;
/* user ID of owner */
    uint32 st_uid;
/* group ID of owner */
    uint32 st_gid;
/* total size, in bytes */
    uint32 st_size;
/* time of last access */
    uint32 st_atime;
/* time of last modification */
    uint32 st_mtime;
/* time of last change */
    uint32 st_ctime;
// name of the directory entry
    char *d_name;
// Constructor and destructor
    SMBdirent();
    ~SMBdirent();
};

This class is used instead of the standard structure dirent for 2 reasons :
    . Portability
    . Increasing performance because all the 'stat' fields are also present in SMB directory listing packets !
