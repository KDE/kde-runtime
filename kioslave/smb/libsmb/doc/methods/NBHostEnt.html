class NBHostEnt
{
public:
// NetBIOS name in NetBIOS mangled format
    char *NBName;
// clear ASCIIZ name
    char *name;
// IP is in host native format
    uint32 ip;
// for group names (not implemented yet)
    NBHostEnt *next;
    NBHostEnt(const char *nbn=0, const char *n=0, uint32 i=0, NBHostEnt *ne=0);
    ~NBHostEnt();
};

Used as a replacement for struct hostent, mainly for need a clean destructor.
